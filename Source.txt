local function Tween(obj, props, info)
    return TweenService:Create(obj, info or TweenInfo.new(TweenTime), props):Play()
end

local function GetMousePosition(obj)
    local x = math.clamp(Mouse.X - obj.AbsolutePosition.X, 0, obj.AbsoluteSize.X)
    local y = math.clamp(Mouse.Y - obj.AbsolutePosition.Y, 0, obj.AbsoluteSize.Y)
    return x, y, x / obj.AbsoluteSize.X, y / obj.AbsoluteSize.Y
end

local function AddHoverScrollbar(sf, color)
    color = color or Color3.fromRGB(80, 80, 90)
    local bar = NewFrame()
    bar.Name = "HoverBar"
    bar.BackgroundColor3 = color
    bar.BackgroundTransparency = 1
    bar.Size = UDim2.new(0, 3, 0, 0)
    bar.Position = UDim2.new(1, -3, 0, 0)
    bar.ZIndex = sf.ZIndex + 5
    bar.Parent = sf
    NewCorner(bar, 2)

    local function Update()
        local canvasY = sf.CanvasSize.Y.Offset
        local windowY = sf.AbsoluteSize.Y
        if canvasY <= windowY then
            bar.Size = UDim2.new(0, 3, 0, 0)
            return
        end
        local ratio = windowY / canvasY
        local barHeight = math.max(20, ratio * windowY)
        local scrollT = sf.CanvasPosition.Y / (canvasY - windowY)
        local barY = scrollT * (windowY - barHeight)
        bar.Size = UDim2.new(0, 3, 0, barHeight)
        bar.Position = UDim2.new(1, -4, 0, barY)
    end

    sf:GetPropertyChangedSignal("CanvasPosition"):Connect(Update)
    sf:GetPropertyChangedSignal("AbsoluteSize"):Connect(Update)
    sf.MouseEnter:Connect(function()
        Tween(bar, {BackgroundTransparency = 0.35}, SlowTweenInfo)
    end)
    sf.MouseLeave:Connect(function()
        Tween(bar, {BackgroundTransparency = 1}, SlowTweenInfo)
    end)
end

        end
        Tween(card, {Position = UDim2.new(0, 260, 0, 0), ImageTransparency = 1}, SlowTweenInfo)

            Tween(card, {Size = UDim2.new(1, 0, 0, 0)}, TweenInfo.new(TweenTime))

        end)
    end

    task.delay(0.02, function()
        Tween(card, {Position = UDim2.new(0, 0, 0, 0)}, SlowTweenInfo)
    end)

    TweenService:Create(fill, TweenInfo.new(duration, Enum.EasingStyle.Linear),
        {Size = UDim2.new(0, 0, 1, 0)}):Play()

    task.delay(duration, notifObj.Dismiss)

    clickBtn.BackgroundTransparency = 1
    clickBtn.Text = ""
    clickBtn.Size = UDim2.new(1, 0, 1, 0)
    clickBtn.ZIndex = 305
    clickBtn.Parent = card
    clickBtn.MouseButton1Click:Connect(notifObj.Dismiss)

    return notifObj

        end)
    end

    return page, layout

    function lib.AddSlider(text, config, callback, parent)
        parent = parent or page
        local min = config.Minimum or config.Min or 0
        local max = config.Maximum or config.Max or 100
        local def = config.Default or config.Def or 0
        if min > max then min, max = max, min end
        def = math.clamp(def, min, max)

        local container, bg = CreateRow(text, parent)
        local btn = NewTextButton(text .. ": " .. def, 12)
        btn.Parent = bg

        local fill = NewRoundedFrame(5)
        fill.Size = UDim2.new((def - min) / (max - min), 0, 1, 0)
        fill.ImageColor3 = Color3.fromRGB(70, 70, 70)
        fill.ImageTransparency = 0.7
        fill.ZIndex = 5
        fill.Parent = btn

        local current = def

        local id = pageTitle .. "_" .. text .. "_slider"
        RegisterConfig(id,
            function() return current end,
            function(v)
                v = math.clamp(v, min, max)
                current = v
                btn.Text = text .. ": " .. v
                Tween(fill, {Size = UDim2.new((v - min) / (max - min), 0, 1, 0)})
                callback(v)
            end
        )

        btn.MouseButton1Down:Connect(function()
            Tween(fill, {ImageTransparency = 0.5})
            local _, _, x = GetMousePosition(btn)
            local val = math.floor(min + (max - min) * x)
            current = val
            callback(val)
            btn.Text = text .. ": " .. val
            Tween(fill, {Size = UDim2.new(x, 0, 1, 0)})

            local moveConn, endConn
            moveConn = Mouse.Move:Connect(function()
                local _, _, x2 = GetMousePosition(btn)
                local v2 = math.floor(min + (max - min) * x2)
                current = v2
                callback(v2)
                btn.Text = text .. ": " .. v2
                Tween(fill, {Size = UDim2.new(x2, 0, 1, 0)})
            end)
            endConn = UserInputService.InputEnded:Connect(function(inp)
                if inp.UserInputType == Enum.UserInputType.MouseButton1 then
                    Tween(fill, {ImageTransparency = 0.7})
                    moveConn:Disconnect()
                    endConn:Disconnect()
                end
            end)
        end)
    end

            end

        local leftBtn = NewTextButton("<", 14)
        leftBtn.Font = Enum.Font.GothamBold
        leftBtn.TextColor3 = Color3.fromRGB(180, 180, 180)
        leftBtn.Size = UDim2.new(0, 18, 1, 0)
        leftBtn.Position = UDim2.new(0.41, 0, 0, 0)
        leftBtn.Parent = container

        local valueLabel = NewTextLabel(options[idx], 11)
        valueLabel.Font = Enum.Font.GothamBold
        valueLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        valueLabel.TextTruncate = Enum.TextTruncate.AtEnd
        valueLabel.Size = UDim2.new(0.36, 0, 1, 0)
        valueLabel.Position = UDim2.new(0.44, 18, 0, 0)
        valueLabel.Parent = container

        local rightBtn = NewTextButton(">", 14)
        rightBtn.Font = Enum.Font.GothamBold
        rightBtn.TextColor3 = Color3.fromRGB(180, 180, 180)
        rightBtn.Size = UDim2.new(0, 18, 1, 0)
        rightBtn.Position = UDim2.new(1, -18, 0, 0)
        rightBtn.Parent = container

        local id = pageTitle .. "_" .. text .. "_stepper"
        RegisterConfig(id,
            function() return idx end,
            function(v)
                idx = math.clamp(v, 1, #options)
                valueLabel.Text = options[idx]
                callback(options[idx], idx)
            end
        )

        local function Step(direction)
            idx = ((idx - 1 + direction) % #options) + 1
            Tween(valueLabel, {TextTransparency = 1}, TweenInfo.new(TweenTime))
            task.delay(TweenTime, function()
                valueLabel.Text = options[idx]
                Tween(valueLabel, {TextTransparency = 0}, TweenInfo.new(TweenTime))
            end)
            Tween(bg, {ImageColor3 = Color3.fromRGB(42, 42, 42)})
            task.wait(TweenTime)
            Tween(bg, {ImageColor3 = Color3.fromRGB(35, 35, 35)})
            callback(options[idx], idx)
        end

        leftBtn.MouseButton1Click:Connect(function() Step(-1) end)
        rightBtn.MouseButton1Click:Connect(function() Step(1) end)

        local function HoverArrow(btn)
            btn.MouseEnter:Connect(function()
                Tween(btn, {TextColor3 = Color3.fromRGB(255, 255, 255)})
            end)
            btn.MouseLeave:Connect(function()
                Tween(btn, {TextColor3 = Color3.fromRGB(180, 180, 180)})
            end)
        end
        HoverArrow(leftBtn)
        HoverArrow(rightBtn)
    end

    return lib
end

local UILibrary = {}

function UILibrary.Load(hubName, gameName, version, topBarConfig)
    hubName = hubName or "Twink UI"
    gameName = gameName or "Unknown Game"
    version = version or "v1.0"
    topBarConfig = topBarConfig or {}

    local targetParent = RunService:IsStudio() and Player:WaitForChild("PlayerGui") or CoreGui

    local old = targetParent:FindFirstChild(hubName)
    if old then old:Destroy() end

    local rootGui = Instance.new("ScreenGui")
    rootGui.Name = hubName
    rootGui.ResetOnSpawn = false
    rootGui.Parent = targetParent

    SetupNotifications(targetParent, hubName)
    SetupWatermark(targetParent, hubName, version)

    local mainGuiVisible = true
    local topBarBtn = SetupTopBar(targetParent, hubName, topBarConfig, function(visible)
        mainGuiVisible = visible
        rootGui.Enabled = visible
    end)

    local container, main, shadow = BuildMainFrame()
    container.Parent = rootGui

    local titleBar = BuildTitleBar(main, hubName, gameName, version, container, shadow)

    local menuBar, display = BuildSidebar(main)

    local userPanel, accentStrip = BuildUserInfo(main)

    local accentObjects = {
        {obj = accentStrip, prop = "BackgroundColor3"},
    }

    local tabCount = 0
    local menuLayout = Instance.new("UIListLayout")
    menuLayout.SortOrder = Enum.SortOrder.LayoutOrder
    menuLayout.Padding = UDim.new(0, 5)
    menuLayout.Parent = menuBar
    menuLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        menuBar.CanvasSize = UDim2.new(0, 0, 0, menuLayout.AbsoluteContentSize.Y + 5)
    end)

    local tabLibrary = {}

    function tabLibrary.AddPage(pageTitle, withSearch)
        withSearch = (withSearch == nil) and true or withSearch

        local isFirst = (tabCount == 0)
        local tabContainer, tabStrip, tabButton = CreateTabButton(menuBar, pageTitle, isFirst)
        local displayPage, pageLayout = CreateDisplayPage(display, pageTitle, isFirst, withSearch)

        table.insert(accentObjects, {obj = tabStrip, prop = "BackgroundColor3"})

        tabButton.MouseButton1Down:Connect(function()
            for _, child in next, menuBar:GetChildren() do
                if child:IsA("GuiObject") and child ~= tabContainer then
                    local isMatch = string.find(child.Name:lower(), pageTitle:lower())
                    local innerBtn = child:FindFirstChild(child.Name .. "Button")
                    local strip = child:FindFirstChild("ActiveStrip")
                    Tween(child, {ImageColor3 = isMatch and Color3.fromRGB(50, 50, 50) or Color3.fromRGB(40, 40, 40)})
                    if innerBtn then Tween(innerBtn, {TextTransparency = isMatch and 0 or 0.45}) end
                    if strip then Tween(strip, {BackgroundTransparency = isMatch and 0 or 1}) end
                end
            end
            for _, child in next, display:GetChildren() do
                if child:IsA("GuiObject") then
                    child.Visible = string.find(child.Name:lower(), pageTitle:lower()) and true or false
                end
            end
        end)

        tabCount = tabCount + 1

        local pageLib = MakePageLibrary(displayPage, display, pageTitle)
        return pageLib
    end

    function tabLibrary.Notify(title, message, notifType, duration)
        if not NotifHolder then return end
        notifType = (notifType or "info"):lower()
        duration = duration or 4

        if #ActiveNotifs >= MaxNotifs then
            local oldest = table.remove(ActiveNotifs, 1)
            if oldest and oldest.Dismiss then oldest.Dismiss() end
        end

        local obj = CreateNotificationCard(title, message, notifType, duration)
        table.insert(ActiveNotifs, obj)
    end

    function tabLibrary.SetWatermark(enabled)
        WatermarkGui.Enabled = enabled == true
    end

    function tabLibrary.SetAccentColor(newColor)
        AccentColor = newColor
        for _, entry in ipairs(accentObjects) do
            pcall(function() entry.obj[entry.prop] = newColor end)
        end
        WMAccent.BackgroundColor3 = newColor
    end

    function tabLibrary.SaveConfig(profileName)
        local data = {}
        for id, entry in pairs(ConfigRegistry) do
            local ok, val = pcall(entry.getter)
            if ok then data[id] = val end
        end
        local ok, encoded = pcall(HttpService.JSONEncode, HttpService, data)
        if not ok then
            warn("[TwinkUI] SaveConfig encode failed:", encoded)
            return
        end
        local filename = hubName .. "_" .. (profileName or "default") .. ".json"
        local sOk, sErr = pcall(writefile, filename, encoded)
        if not sOk then warn("[TwinkUI] SaveConfig writefile failed:", sErr) end
    end

    function tabLibrary.LoadConfig(profileName)
        local filename = hubName .. "_" .. (profileName or "default") .. ".json"
        local ok, raw = pcall(readfile, filename)
        if not ok then
            warn("[TwinkUI] LoadConfig: file not found:", filename)
            return
        end
        local dOk, data = pcall(HttpService.JSONDecode, HttpService, raw)
        if not dOk then
            warn("[TwinkUI] LoadConfig decode failed:", data)
            return
        end
        for id, val in pairs(data) do
            if ConfigRegistry[id] then
                pcall(ConfigRegistry[id].setter, val)
            end
        end
    end

    return tabLibrary
end

return UILibrary
