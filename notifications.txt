local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")
local TextService = game:GetService("TextService")

local CoreGui = game:GetService("CoreGui")

local NotificationGui = Instance.new("ScreenGui")
NotificationGui.Name = "AkaliNotif"
NotificationGui.Parent = CoreGui

local Container = Instance.new("Frame")
Container.Name = "Container"
Container.Position = UDim2.new(0, 20, 0.5, -20)
Container.Size = UDim2.new(0, 300, 0.5, 0)
Container.BackgroundTransparency = 1
Container.Parent = NotificationGui

local Padding = 10
local DescriptionPadding = 10
local TweenTime = 1
local TweenStyle = Enum.EasingStyle.Sine
local TweenDirection = Enum.EasingDirection.Out

local ActiveNotifications = {}
local LastTick = tick()

local function CreateRoundedBackground()
	local Background = Instance.new("ImageLabel")
	Background.Image = "http://www.roblox.com/asset/?id=5761488251"
	Background.ScaleType = Enum.ScaleType.Slice
	Background.SliceCenter = Rect.new(2, 2, 298, 298)
	Background.ImageColor3 = Color3.fromRGB(30, 30, 30)
	Background.BackgroundTransparency = 1
	return Background
end

local function CreateShadow()
	local Shadow = Instance.new("ImageLabel")
	Shadow.Image = "http://www.roblox.com/asset/?id=5761498316"
	Shadow.ScaleType = Enum.ScaleType.Slice
	Shadow.SliceCenter = Rect.new(17, 17, 283, 283)
	Shadow.Size = UDim2.fromScale(1, 1) + UDim2.fromOffset(30, 30)
	Shadow.Position = -UDim2.fromOffset(15, 15)
	Shadow.ImageColor3 = Color3.fromRGB(30, 30, 30)
	Shadow.BackgroundTransparency = 1
	return Shadow
end

local function CreateTextLabel(Text, Font, Size, IsButton)
	local Label = Instance.new(IsButton and "TextButton" or "TextLabel")
	Label.Text = Text
	Label.Font = Font
	Label.TextSize = Size
	Label.BackgroundTransparency = 1
	Label.TextXAlignment = Enum.TextXAlignment.Left
	Label.RichText = true
	Label.TextColor3 = Color3.fromRGB(255, 255, 255)
	return Label
end

local function CreateTitleLabel(Text)
	return CreateTextLabel(Text, Enum.Font.GothamSemibold, 14)
end

local function CreateDescriptionLabel(Text)
	return CreateTextLabel(Text, Enum.Font.Gotham, 14)
end

local function FadeOutObject(Object)
	local function Fade(Instance)
		local ClassName = Instance.ClassName
		local Property
		if ClassName:sub(1, 4) == "Text" then
			Property = "TextTransparency"
		elseif ClassName:sub(1, 4) == "Fram" then
			Property = "BackgroundTransparency"
		elseif ClassName:sub(1, 4) == "Imag" then
			Property = "ImageTransparency"
		else
			return
		end
		TweenService:Create(Instance, TweenInfo.new(0.25, TweenStyle, TweenDirection), {
			[Property] = 1
		}):Play()
	end
	Fade(Object)
	for _, Child in ipairs(Object:GetDescendants()) do
		Fade(Child)
	end
end

local function RemoveNotificationAfterDelay(Entry)
	local Background, Duration = Entry[1], Entry[4]
	task.wait(Duration)
	FadeOutObject(Background)
	task.wait(0.25)
	Background.Parent = nil

	for Index, ExistingEntry in ipairs(ActiveNotifications) do
		if ExistingEntry == Entry then
			table.remove(ActiveNotifications, Index)
			break
		end
	end

	for _, ExistingEntry in ipairs(ActiveNotifications) do
		ExistingEntry[2] = 0
		ExistingEntry[3] = false
	end
end

local function UpdatePositions()
	local Delta = tick() - LastTick
	local OffsetY = 0
	for _, Entry in ipairs(ActiveNotifications) do
		local Label, Progress, Done, _, Height = Entry[1], Entry[2], Entry[3], Entry[4], Entry[5]
		if not Done then
			Progress = math.min(Progress + Delta / TweenTime, 1)
			Entry[2] = Progress
			if Progress >= 1 then
				Entry[3] = true
				Done = true
			end
		end
		local EasedProgress = TweenService:GetValue(Progress, TweenStyle, TweenDirection)
		local TargetY = OffsetY
		local CurrentPos = Label.Position
		local TargetPos = UDim2.new(0, 0, 0, TargetY)
		Label.Position = CurrentPos:Lerp(TargetPos, EasedProgress)
		OffsetY = OffsetY + Height + Padding
	end
	LastTick = tick()
end

RunService:BindToRenderStep("UpdateList", 0, UpdatePositions)

return {
	Notify = function(Properties)
		Properties = Properties or {}
		local Title = Properties.Title
		local Description = Properties.Description
		local Duration = Properties.Duration or 5

		if not (Title or Description) then
			return
		end

		local Height = 0
		if Title then
			Height = Height + 26
		end
		if Description then
			local MaxWidth = Container.AbsoluteSize.X - Padding - DescriptionPadding
			local TextSize = TextService:GetTextSize(Description, 14, Enum.Font.Gotham, Vector2.new(0, 0))
			local LineCount = math.max(1, math.ceil(TextSize.X / MaxWidth))
			Height = Height + (TextSize.Y * LineCount) + 8
		end

		local Background = CreateRoundedBackground()
		Background.Size = UDim2.new(1, 0, 0, Height)

		local TotalHeight = 0
		for _, Entry in ipairs(ActiveNotifications) do
			TotalHeight = TotalHeight + Entry[5] + Padding
		end
		Background.Position = UDim2.new(-1, 20, 0, TotalHeight)

		CreateShadow().Parent = Background

		if Title then
			local TitleLabel = CreateTitleLabel(Title)
			TitleLabel.Size = UDim2.new(1, -10, 0, 26)
			TitleLabel.Position = UDim2.fromOffset(10, 0)
			TitleLabel.Parent = Background
		end

		if Description then
			local DescLabel = CreateDescriptionLabel(Description)
			DescLabel.TextWrapped = true
			DescLabel.Size = UDim2.fromScale(1, 1) + UDim2.fromOffset(-DescriptionPadding, Title and -26 or 0)
			DescLabel.Position = UDim2.fromOffset(10, Title and 26 or 0)
			DescLabel.TextYAlignment = Title and Enum.TextYAlignment.Top or Enum.TextYAlignment.Center
			DescLabel.Parent = Background
		end

		Background.Parent = Container

		local Entry = { Background, 0, false, Duration, Height }
		table.insert(ActiveNotifications, Entry)

		task.spawn(RemoveNotificationAfterDelay, Entry)
	end
}
