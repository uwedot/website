    Keybind = Enum.KeyCode.RightControl,
    AccentColor = Color3.fromRGB(60, 180, 255),

local function CreateInstance(ClassName, Properties)
    local Instance = Instance.new(ClassName)
    for Key, Value in pairs(Properties) do
        Instance[Key] = Value
    end
    return Instance
end

local function AddCorner(Parent, Radius)
    CreateInstance("UICorner", { Parent = Parent, CornerRadius = UDim.new(0, Radius) })
end

local function CreateRipple(Button)
    Button.ClipsDescendants = true
    Button.MouseButton1Down:Connect(function(Input)
        task.spawn(function()
            local Ripple = CreateInstance("ImageLabel", {
                Parent = Button,
                Name = "Ripple",
                BackgroundTransparency = 1,
                BorderSizePixel = 0,
                Image = "rbxassetid://2708891598",
                ImageColor3 = Color3.fromRGB(255, 255, 255),
                ImageTransparency = 0.8,
                ZIndex = 9,
            })
            local X = Input.Position.X - Button.AbsolutePosition.X
            local Y = Input.Position.Y - Button.AbsolutePosition.Y
            Ripple.Position = UDim2.new(0, X, 0, Y)
            Ripple.Size = UDim2.new(0, 0, 0, 0)
            TweenService:Create(Ripple, TweenInfo.new(0.4), {
                Size = UDim2.new(0, 500, 0, 500),
                Position = UDim2.new(0, X - 250, 0, Y - 250)
            }):Play()
            TweenService:Create(Ripple, TweenInfo.new(0.4), { ImageTransparency = 1 }):Play()
            task.wait(0.4)
            Ripple:Destroy()
        end)
    end)
end

local function GetMousePosition()
    if UserInputService:GetMouseLocation then
        return UserInputService:GetMouseLocation()
    end
    return LocalPlayer:GetMouse().Position
end

function Borealis.New(Title, Config)
    Config = Config or {}
    local Theme = {
        Background = Config.Background or DefaultConfig.Background,
        Accent = Config.AccentColor or DefaultConfig.AccentColor,
        LightContrast = Config.LightContrast or DefaultConfig.LightContrast,
        DarkContrast = Config.DarkContrast or DefaultConfig.DarkContrast,
        TextColor = Config.TextColor or DefaultConfig.TextColor,
    }
    local Keybind = Config.Keybind or DefaultConfig.Keybind
    local UseFloatingToggle = (Config.FloatingToggle ~= nil) and Config.FloatingToggle or DefaultConfig.FloatingToggle

    if CoreGui:FindFirstChild("BorealisUI") then
        CoreGui.BorealisUI:Destroy()
    end

    local ScreenGui = CreateInstance("ScreenGui", {
        Name = "BorealisUI",
        Parent = CoreGui,
        ResetOnSpawn = false,
    })

    local ThemeUpdaters = {}
    local ToggleButton = nil

    if UseFloatingToggle then
        ToggleButton = CreateInstance("TextButton", {
            Parent = ScreenGui,
            BackgroundColor3 = Theme.LightContrast,
            Position = UDim2.new(0.1, 0, 0.1, 0),
            Size = UDim2.new(0, 50, 0, 50),
            Font = Enum.Font.GothamBold,
            Text = "UI",
            TextColor3 = Theme.Accent,
            TextSize = 20,
            Active = true,
            Draggable = true,
            ZIndex = 10,
            Visible = true,
        })
        AddCorner(ToggleButton, 50)
        local Stroke = CreateInstance("UIStroke", {
            Parent = ToggleButton,
            Thickness = 2,
            Color = Theme.Accent,
            ApplyStrokeMode = Enum.ApplyStrokeMode.Border,
        })
        table.insert(ThemeUpdaters, function(Color)
            ToggleButton.TextColor3 = Color
            Stroke.Color = Color
        end)
    end

    local MainFrame = CreateInstance("Frame", {
        Parent = ScreenGui,
        Name = "Main",
        BackgroundColor3 = Theme.Background,
        BorderSizePixel = 0,
        Position = UDim2.new(0.5, -275, 0.5, -175),
        Size = UDim2.new(0, 550, 0, 350),
        ClipsDescendants = true,
        Visible = not UseFloatingToggle,
    })
    AddCorner(MainFrame, 6)
    local MainStroke = CreateInstance("UIStroke", {
        Parent = MainFrame,
        Thickness = 1,
        Color = Theme.Accent,
        Transparency = UseFloatingToggle and 1 or 0,
        ApplyStrokeMode = Enum.ApplyStrokeMode.Border,
    })
    table.insert(ThemeUpdaters, function(Color) MainStroke.Color = Color end)

    local function FadeOutAndDestroy()
        local CloseTween = TweenService:Create(MainFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
            Size = UDim2.new(0, 0, 0, 0),
            BackgroundTransparency = 1,
        })
        CloseTween:Play()
        for _, Child in pairs(MainFrame:GetDescendants()) do
            if Child:IsA("GuiObject") then
                TweenService:Create(Child, TweenInfo.new(0.3), { BackgroundTransparency = 1 }):Play()
            end
            if Child:IsA("TextLabel") or Child:IsA("TextButton") or Child:IsA("TextBox") then
                TweenService:Create(Child, TweenInfo.new(0.3), { TextTransparency = 1 }):Play()
            end
        end
        CloseTween.Completed:Wait()
        ScreenGui:Destroy()
        if ToggleButton then ToggleButton:Destroy() end
    end

    local TopBar = CreateInstance("Frame", {
        Parent = MainFrame,
        BackgroundColor3 = Theme.LightContrast,
        BorderSizePixel = 0,
        Size = UDim2.new(1, 0, 0, 40),
    })
    AddCorner(TopBar, 6)
    CreateInstance("Frame", {
        Parent = TopBar,
        BackgroundColor3 = Theme.LightContrast,
        BorderSizePixel = 0,
        Position = UDim2.new(0, 0, 1, -5),
        Size = UDim2.new(1, 0, 0, 5),
    })

    local TitleLabel = CreateInstance("TextLabel", {
        Parent = TopBar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 15, 0, 0),
        Size = UDim2.new(0, 200, 1, 0),
        Font = Enum.Font.GothamBold,
        Text = Title,
        TextColor3 = Theme.TextColor,
        TextSize = 16,
        TextXAlignment = Enum.TextXAlignment.Left,
        ZIndex = 2,
    })

    local SearchButton = CreateInstance("ImageButton", {
        Parent = TopBar,
        BackgroundTransparency = 1,
        Position = UDim2.new(1, -62, 0, 10),
        Size = UDim2.new(0, 20, 0, 20),
        Image = "rbxassetid://6031154871",
        ImageColor3 = Color3.fromRGB(150, 150, 150),
        ZIndex = 2,
    })

    local SearchBar = CreateInstance("TextBox", {
        Parent = TopBar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 15, 0, 0),
        Size = UDim2.new(1, -100, 1, 0),
        Font = Enum.Font.GothamBold,
        Text = "",
        TextColor3 = Theme.TextColor,
        TextSize = 16,
        TextXAlignment = Enum.TextXAlignment.Left,
        PlaceholderText = "Search...",
        PlaceholderColor3 = Color3.fromRGB(150, 150, 150),
        Visible = false,
        ZIndex = 2,
        ClearTextOnFocus = false,
    })

    local CloseButton = CreateInstance("TextButton", {
        Parent = TopBar,
        BackgroundTransparency = 1,
        Position = UDim2.new(1, -35, 0, 5),
        Size = UDim2.new(0, 30, 0, 30),
        Font = Enum.Font.GothamBold,
        Text = "X",
        TextColor3 = Color3.fromRGB(150, 150, 150),
        TextSize = 16,
        ZIndex = 2,
    })
    CloseButton.MouseButton1Click:Connect(FadeOutAndDestroy)

    local IsDragging, DragInput, DragStart, StartPos
    local function UpdateDrag(Input)
        local Delta = Input.Position - DragStart
        TweenService:Create(MainFrame, TweenInfo.new(0.05), {
            Position = UDim2.new(StartPos.X.Scale, StartPos.X.Offset + Delta.X,
                                 StartPos.Y.Scale, StartPos.Y.Offset + Delta.Y)
        }):Play()
    end

    TopBar.InputBegan:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
            IsDragging = true
            DragStart = Input.Position
            StartPos = MainFrame.Position
        end
    end)
    TopBar.InputChanged:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch then
            DragInput = Input
        end
    end)
    UserInputService.InputChanged:Connect(function(Input)
        if Input == DragInput and IsDragging then UpdateDrag(Input) end
    end)
    UserInputService.InputEnded:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
            IsDragging = false
        end
    end)

    local Resizer = CreateInstance("ImageButton", {
        Parent = MainFrame,
        BackgroundTransparency = 1,
        Position = UDim2.new(1, -20, 1, -20),
        Size = UDim2.new(0, 20, 0, 20),
        Image = "rbxassetid://3926307971",
        ImageRectOffset = Vector2.new(204, 364),
        ImageRectSize = Vector2.new(36, 36),
        ImageColor3 = Theme.Accent,
        ZIndex = 10,
    })
    table.insert(ThemeUpdaters, function(Color) Resizer.ImageColor3 = Color end)

    local IsResizing, ResizeStartPos, StartSize
    Resizer.InputBegan:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
            IsResizing = true
            ResizeStartPos = Input.Position
            StartSize = MainFrame.Size
        end
    end)
    UserInputService.InputChanged:Connect(function(Input)
        if IsResizing and (Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch) then
            local Delta = Input.Position - ResizeStartPos
            local NewX = math.max(350, StartSize.X.Offset + Delta.X)
            local NewY = math.max(200, StartSize.Y.Offset + Delta.Y)
            MainFrame.Size = UDim2.new(0, NewX, 0, NewY)
        end
    end)
    UserInputService.InputEnded:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
            IsResizing = false
        end
    end)

    local Sidebar = CreateInstance("Frame", {
        Parent = MainFrame,
        BackgroundColor3 = Theme.LightContrast,
        BorderSizePixel = 0,
        Position = UDim2.new(0, 0, 0, 40),
        Size = UDim2.new(0, 130, 1, -40),
    })
    AddCorner(Sidebar, 6)
    CreateInstance("Frame", {
        Parent = Sidebar,
        BackgroundColor3 = Theme.LightContrast,
        BorderSizePixel = 0,
        Position = UDim2.new(0, 0, 0, 0),
        Size = UDim2.new(1, 0, 0, 10),
    })

    local TabContainer = CreateInstance("ScrollingFrame", {
        Parent = Sidebar,
        BackgroundTransparency = 1,
        Size = UDim2.new(1, 0, 1, -10),
        Position = UDim2.new(0, 0, 0, 10),
        ScrollBarThickness = 0,
        ScrollingEnabled = false,
    })
    CreateInstance("UIListLayout", {
        Parent = TabContainer,
        SortOrder = Enum.SortOrder.LayoutOrder,
        Padding = UDim.new(0, 5),
    })

    local PagesFrame = CreateInstance("Frame", {
        Parent = MainFrame,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 140, 0, 50),
        Size = UDim2.new(1, -150, 1, -60),
        ClipsDescendants = true,
    })

    local TabList = {}
    local CurrentTab = nil
    local IsFirstTab = true

    local function CreateTab(Name)
        local TabButton = CreateInstance("TextButton", {
            Parent = TabContainer,
            BackgroundColor3 = Theme.LightContrast,
            BorderSizePixel = 0,
            Size = UDim2.new(1, 0, 0, 30),
            Font = Enum.Font.GothamMedium,
            Text = Name,
            TextColor3 = Color3.fromRGB(150, 150, 150),
            TextSize = 13,
            AutoButtonColor = false,
        })
        local Indicator = CreateInstance("Frame", {
            Parent = TabButton,
            BackgroundColor3 = Theme.Accent,
            BorderSizePixel = 0,
            AnchorPoint = Vector2.new(0, 0.5),
            Position = UDim2.new(0, 0, 0.5, 0),
            Size = UDim2.new(0, 3, 0, 0),
            Visible = true,
        })
        table.insert(ThemeUpdaters, function(Color) Indicator.BackgroundColor3 = Color end)

        local Page = CreateInstance("ScrollingFrame", {
            Parent = PagesFrame,
            Name = Name .. "Page",
            BackgroundTransparency = 1,
            Size = UDim2.new(1, 0, 1, 0),
            ScrollBarThickness = 0,
            ScrollBarImageColor3 = Theme.Accent,
            Visible = false,
            CanvasSize = UDim2.new(0, 0, 0, 0),
        })
        table.insert(ThemeUpdaters, function(Color) Page.ScrollBarImageColor3 = Color end)
        CreateInstance("UIPadding", {
            Parent = Page,
            PaddingTop = UDim.new(0, 12),
            PaddingLeft = UDim.new(0, 12),
            PaddingRight = UDim.new(0, 12),
            PaddingBottom = UDim.new(0, 12),
        })
        local PageLayout = CreateInstance("UIListLayout", {
            Parent = Page,
            SortOrder = Enum.SortOrder.LayoutOrder,
            Padding = UDim.new(0, 12),
        })
        PageLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            Page.CanvasSize = UDim2.new(0, 0, 0, PageLayout.AbsoluteContentSize.Y + 24)
        end)

        local function SwitchTab()
            if CurrentTab == Page then return end
            if SearchBar.Visible then SearchBar.Text = "" end

            for _, TabEntry in pairs(TabList) do
                if TabEntry.Page.Visible then
                    TweenService:Create(TabEntry.Button, TweenHover, { TextColor3 = Color3.fromRGB(150, 150, 150) }):Play()
                    TweenService:Create(TabEntry.Indicator, TweenLine, { Size = UDim2.new(0, 3, 0, 0) }):Play()
                    task.spawn(function()
                        local OldPage = TabEntry.Page
                        for _, Child in pairs(OldPage:GetDescendants()) do
                            if Child:IsA("GuiObject") then
                                TweenService:Create(Child, TweenInfo.new(0.2), { BackgroundTransparency = 1 }):Play()
                            end
                            if Child:IsA("TextLabel") or Child:IsA("TextButton") or Child:IsA("TextBox") then
                                TweenService:Create(Child, TweenInfo.new(0.2), { TextTransparency = 1 }):Play()
                            end
                        end
                        OldPage.Visible = false
                    end)
                end
            end

            CurrentTab = Page
            Page.Visible = true
            TweenService:Create(Indicator, TweenLine, { Size = UDim2.new(0, 3, 1, 0) }):Play()

            Page.Position = UDim2.new(0, 0, 0, 15)
            for _, Child in pairs(Page:GetChildren()) do
                if Child:IsA("GuiObject") then Child.BackgroundTransparency = 1 end
                if Child:IsA("TextLabel") or Child:IsA("TextButton") or Child:IsA("TextBox") then
                    Child.TextTransparency = 1
                end
            end
            TweenService:Create(Page, TweenTab, { Position = UDim2.new(0, 0, 0, 0) }):Play()
            TweenService:Create(TabButton, TweenHover, { TextColor3 = Theme.TextColor }):Play()

            for _, Child in pairs(Page:GetDescendants()) do
                if Child:IsA("TextLabel") or Child:IsA("TextButton") or Child:IsA("TextBox") then
                    TweenService:Create(Child, TweenInfo.new(0.3), { TextTransparency = 0 }):Play()
                elseif Child:IsA("Frame") and Child.Name ~= "Shadow" and Child.Name ~= "Ripple" then
                    if Child.BackgroundColor3 == Theme.LightContrast or
                       Child.BackgroundColor3 == Theme.DarkContrast or
                       Child.BackgroundColor3 == Theme.Accent then
                        TweenService:Create(Child, TweenInfo.new(0.3), { BackgroundTransparency = 0 }):Play()
                    end
                    Child.Visible = true
                end
                if Child:IsA("GuiObject") and Child:FindFirstChild("UICorner") then
                    TweenService:Create(Child, TweenInfo.new(0.3), { BackgroundTransparency = 0 }):Play()
                end
            end
        end

        TabButton.MouseButton1Click:Connect(SwitchTab)

        if IsFirstTab then
            IsFirstTab = false
            CurrentTab = Page
            Page.Visible = true
            TabButton.TextColor3 = Theme.TextColor
            Indicator.Size = UDim2.new(0, 3, 1, 0)
        end

        table.insert(TabList, { Button = TabButton, Page = Page, Indicator = Indicator })
        return Page, SwitchTab
    end

    local function CreateRow(Parent, Height)
        local Frame = CreateInstance("Frame", {
            Parent = Parent,
            BackgroundColor3 = Theme.LightContrast,
            BorderSizePixel = 0,
            Size = UDim2.new(1, 0, 0, Height),
        })
        AddCorner(Frame, 4)
        return Frame
    end

    local function CreateLabelInside(Row, Text, Offset, SizeX)
        return CreateInstance("TextLabel", {
            Parent = Row,
            BackgroundTransparency = 1,
            Position = UDim2.new(0, Offset or 10, 0, 0),
            Size = UDim2.new(1, -(SizeX or 50), 1, 0),
            Font = Enum.Font.Gotham,
            Text = Text,
            TextColor3 = Theme.TextColor,
            TextSize = 13,
            TextXAlignment = Enum.TextXAlignment.Left,
        })
    end

    local Library = {}

    function Library:UpdateTheme(NewColor)
        Theme.Accent = NewColor
        for _, Updater in pairs(ThemeUpdaters) do Updater(NewColor) end
    end

    function Library:Unload()
        FadeOutAndDestroy()
    end

    local function ToggleUI()
        if MainFrame.Visible then
            local Tween = TweenService:Create(MainFrame, TweenInfo.new(0.2), { Size = UDim2.new(0, 0, 0, 0) })
            Tween:Play()
            Tween.Completed:Wait()
            MainFrame.Visible = false
            MainFrame.ClipsDescendants = true
            TweenService:Create(MainStroke, TweenInfo.new(0.3), { Transparency = 1 }):Play()
        else
            MainFrame.Visible = true
            TweenService:Create(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Back), { Size = UDim2.new(0, 550, 0, 350) }):Play()
            TweenService:Create(MainStroke, TweenInfo.new(0.3), { Transparency = 0 }):Play()
        end
    end

    function Library:Toggle()
        ToggleUI()
    end

    UserInputService.InputBegan:Connect(function(Input, Processed)
        if not Processed and Input.KeyCode == Keybind then ToggleUI() end
    end)

    if ToggleButton then
        ToggleButton.MouseButton1Click:Connect(ToggleUI)
    end

    local SearchOpen = false
    SearchButton.MouseButton1Click:Connect(function()
        SearchOpen = not SearchOpen
        if SearchOpen then
            TitleLabel.Visible = false
            SearchBar.Visible = true
            SearchBar:CaptureFocus()
            TweenService:Create(SearchButton, TweenInfo.new(0.2), { ImageColor3 = Theme.TextColor }):Play()
        else
            TitleLabel.Visible = true
            SearchBar.Visible = false
            SearchBar.Text = ""
            TweenService:Create(SearchButton, TweenInfo.new(0.2), { ImageColor3 = Color3.fromRGB(150, 150, 150) }):Play()
            if CurrentTab then
                for _, Child in pairs(CurrentTab:GetChildren()) do
                    if Child:IsA("Frame") then Child.Visible = true end
                end
            end
        end
    end)

    SearchBar:GetPropertyChangedSignal("Text"):Connect(function()
        local Input = SearchBar.Text:lower()
        if CurrentTab then
            for _, Child in pairs(CurrentTab:GetChildren()) do
                if Child:IsA("Frame") then
                    local Label = Child:FindFirstChild("TextLabel")
                    if Label then
                        Child.Visible = string.find(Label.Text:lower(), Input) ~= nil
                    end
                end
            end
        end
    end)

    function Library:MakeTab(Name)
        local Page, _ = CreateTab(Name)

        local Section = {}

        function Section:AddSection() return Section end

        function Section:AddLabel(Text)
            local Row = CreateRow(Page, 30)
            CreateInstance("TextLabel", {
                Parent = Row,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 10, 0, 0),
                Size = UDim2.new(1, -20, 1, 0),
                Font = Enum.Font.GothamBold,
                Text = Text,
                TextColor3 = Theme.TextColor,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
            })
        end

        function Section:AddTextbox(Props)
            local Row = CreateRow(Page, 36)
            CreateLabelInside(Row, Props.Name, 10, 160)

            local InputBox = CreateInstance("TextBox", {
                Parent = Row,
                BackgroundColor3 = Theme.DarkContrast,
                BorderSizePixel = 0,
                Position = UDim2.new(1, -150, 0.5, -12),
                Size = UDim2.new(0, 140, 0, 24),
                Font = Enum.Font.Gotham,
                Text = Props.Default or "",
                TextColor3 = Color3.fromRGB(200, 200, 200),
                TextSize = 12,
                PlaceholderText = Props.Placeholder or "Type here...",
                ClearTextOnFocus = false,
            })
            AddCorner(InputBox, 4)

            InputBox.Focused:Connect(function()
                TweenService:Create(InputBox, TweenInfo.new(0.2), { TextColor3 = Theme.Accent }):Play()
            end)
            InputBox.FocusLost:Connect(function(EnterPressed)
                TweenService:Create(InputBox, TweenInfo.new(0.2), { TextColor3 = Color3.fromRGB(200, 200, 200) }):Play()
                if Props.Callback then Props.Callback(InputBox.Text) end
                if Props.ClearOnFocusLost then InputBox.Text = "" end
            end)
            table.insert(ThemeUpdaters, function(Color)
                if InputBox:IsFocused() then InputBox.TextColor3 = Color end
            end)
        end

        function Section:AddToggle(Props)
            local Row = CreateRow(Page, 36)
            CreateLabelInside(Row, Props.Name, 10, 50)

            local Checkbox = CreateInstance("TextButton", {
                Parent = Row,
                BackgroundColor3 = Theme.DarkContrast,
                BorderSizePixel = 0,
                Position = UDim2.new(1, -30, 0.5, -10),
                Size = UDim2.new(0, 20, 0, 20),
                Text = "",
                AutoButtonColor = false,
            })
            AddCorner(Checkbox, 4)
            CreateRipple(Checkbox)

            local State = Props.Default or false
            local function UpdateToggle()
                TweenService:Create(Checkbox, TweenHover, {
                    BackgroundColor3 = State and Theme.Accent or Theme.DarkContrast
                }):Play()
            end
            UpdateToggle()
            table.insert(ThemeUpdaters, function(Color)
                if State then Checkbox.BackgroundColor3 = Color end
            end)

            Checkbox.MouseButton1Click:Connect(function()
                State = not State
                UpdateToggle()
                if Props.Callback then Props.Callback(State) end
            end)
        end

        function Section:AddButton(Props)
            local Button = CreateInstance("TextButton", {
                Parent = Page,
                BackgroundColor3 = Theme.LightContrast,
                BorderSizePixel = 0,
                Size = UDim2.new(1, 0, 0, 36),
                Font = Enum.Font.Gotham,
                Text = Props.Name,
                TextColor3 = Theme.TextColor,
                TextSize = 13,
                AutoButtonColor = false,
            })
            AddCorner(Button, 4)
            CreateRipple(Button)

            Button.MouseEnter:Connect(function()
                TweenService:Create(Button, TweenHover, { BackgroundColor3 = Color3.fromRGB(45, 45, 45) }):Play()
            end)
            Button.MouseLeave:Connect(function()
                TweenService:Create(Button, TweenHover, { BackgroundColor3 = Theme.LightContrast }):Play()
            end)
            Button.MouseButton1Click:Connect(function()
                if Props.Callback then Props.Callback() end
            end)
        end

        function Section:AddSlider(Props)
            local Row = CreateRow(Page, 50)
            local Label = CreateLabelInside(Row, Props.Name, 10, 20)
            Label.Size = UDim2.new(1, -20, 0, 20)
            Label.Position = UDim2.new(0, 10, 0, 5)

            local ValueLabel = CreateInstance("TextLabel", {
                Parent = Row,
                BackgroundTransparency = 1,
                Position = UDim2.new(1, -60, 0, 5),
                Size = UDim2.new(0, 50, 0, 20),
                Font = Enum.Font.GothamBold,
                Text = tostring(Props.Default or Props.Min),
                TextColor3 = Theme.TextColor,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Right,
            })

            local Min, Max, Default = Props.Min or 0, Props.Max or 100, Props.Default or 0
            local Track = CreateInstance("TextButton", {
                Parent = Row,
                BackgroundColor3 = Theme.DarkContrast,
                BorderSizePixel = 0,
                Position = UDim2.new(0, 10, 0, 32),
                Size = UDim2.new(1, -20, 0, 4),
                Text = "",
                AutoButtonColor = false,
            })
            AddCorner(Track, 2)
            local Fill = CreateInstance("Frame", {
                Parent = Track,
                BackgroundColor3 = Theme.Accent,
                BorderSizePixel = 0,
                Size = UDim2.new((Default - Min) / (Max - Min), 0, 1, 0),
            })
            AddCorner(Fill, 2)
            table.insert(ThemeUpdaters, function(Color) Fill.BackgroundColor3 = Color end)

            local IsDraggingSlider = false
            local function UpdateSlider(Input)
                local Percent = math.clamp((Input.Position.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
                local Value = math.floor(Min + (Max - Min) * Percent)
                TweenService:Create(Fill, TweenInfo.new(0.1), { Size = UDim2.new(Percent, 0, 1, 0) }):Play()
                ValueLabel.Text = tostring(Value)
                if Props.Callback then Props.Callback(Value) end
            end

            Track.InputBegan:Connect(function(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    IsDraggingSlider = true
                    UpdateSlider(Input)
                end
            end)
            UserInputService.InputChanged:Connect(function(Input)
                if IsDraggingSlider and (Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch) then
                    UpdateSlider(Input)
                end
            end)
            UserInputService.InputEnded:Connect(function(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    IsDraggingSlider = false
                end
            end)
        end

        function Section:AddColorPicker(Props)
            local Row = CreateRow(Page, 36)
            Row.ClipsDescendants = true
            local Label = CreateLabelInside(Row, Props.Name, 10, 60)

            local ColorDisplay = CreateInstance("Frame", {
                Parent = Row,
                BackgroundColor3 = Props.Default or Theme.Accent,
                BorderSizePixel = 0,
                Position = UDim2.new(1, -45, 0, 8),
                Size = UDim2.new(0, 35, 0, 20),
            })
            AddCorner(ColorDisplay, 4)

            local Trigger = CreateInstance("TextButton", {
                Parent = Row,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, 36),
                Text = "",
            })
            CreateRipple(Trigger)

            local PickerContainer = CreateInstance("Frame", {
                Parent = Row,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 10, 0, 40),
                Size = UDim2.new(1, -20, 0, 110),
            })

            local ColorSquare = CreateInstance("ImageLabel", {
                Parent = PickerContainer,
                BackgroundColor3 = Color3.fromRGB(255, 0, 4),

                Size = UDim2.new(1, -40, 1, 0),

            local ColorSelection = CreateInstance("ImageLabel", {
                Parent = ColorSquare,
                BackgroundTransparency = 1,
                Size = UDim2.new(0, 18, 0, 18),
                Image = "http://www.roblox.com/asset/?id=4805639000",
                AnchorPoint = Vector2.new(0.5, 0.5),
                Position = UDim2.new(0, 0, 0, 0),
            })

            local HueStrip = CreateInstance("ImageLabel", {
                Parent = PickerContainer,
                BackgroundColor3 = Color3.fromRGB(255, 255, 255),
                BorderSizePixel = 0,
                Position = UDim2.new(1, -25, 0, 0),
                Size = UDim2.new(0, 25, 0, 100),
                Active = true,
            })
            AddCorner(HueStrip, 4)
            CreateInstance("UIGradient", {
                Parent = HueStrip,
                Rotation = 270,
                Color = ColorSequence.new({
                    ColorSequenceKeypoint.new(0.00, Color3.fromRGB(255, 0, 4)),
                    ColorSequenceKeypoint.new(0.20, Color3.fromRGB(234, 255, 0)),
                    ColorSequenceKeypoint.new(0.40, Color3.fromRGB(21, 255, 0)),
                    ColorSequenceKeypoint.new(0.60, Color3.fromRGB(0, 255, 255)),
                    ColorSequenceKeypoint.new(0.80, Color3.fromRGB(0, 17, 255)),
                    ColorSequenceKeypoint.new(0.90, Color3.fromRGB(255, 0, 251)),
                    ColorSequenceKeypoint.new(1.00, Color3.fromRGB(255, 0, 4)),
                })
            })

            local HueSelection = CreateInstance("ImageLabel", {
                Parent = HueStrip,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, 10),
                Image = "rbxassetid://4805639000",
                AnchorPoint = Vector2.new(0.5, 0.5),
                Position = UDim2.new(0.5, 0, 0, 0),
            })

            local RainbowButton = CreateInstance("TextButton", {
                Parent = PickerContainer,
                BackgroundColor3 = Theme.DarkContrast,
                BorderSizePixel = 0,
                Position = UDim2.new(0, 210, 0, 0),
                Size = UDim2.new(0, 80, 0, 20),
                Font = Enum.Font.Gotham,
                Text = "Rainbow",
                TextColor3 = Color3.fromRGB(200, 200, 200),
                TextSize = 11,
            })
            AddCorner(RainbowButton, 4)

            local ColorH, ColorS, ColorV = 1, 1, 1
            local RainbowMode = false
            local RainbowValue = 0
            local HuePosition = 0
            local ColorInputConnection, HueInputConnection

            local function UpdatePicker()
                ColorDisplay.BackgroundColor3 = Color3.fromHSV(ColorH, ColorS, ColorV)
                ColorSquare.BackgroundColor3 = Color3.fromHSV(ColorH, 1, 1)
                if Props.Callback then Props.Callback(ColorDisplay.BackgroundColor3) end
                Library:UpdateTheme(ColorDisplay.BackgroundColor3)
            end

            if Props.Default then
                local H, S, V = Color3.toHSV(Props.Default)
                ColorH, ColorS, ColorV = H, S, V
                ColorSelection.Position = UDim2.new(S, 0, 1 - V, 0)
                HueSelection.Position = UDim2.new(0.5, 0, 1 - H, 0)
                UpdatePicker()
            end

            local function OnColorSquareInput(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    if RainbowMode then return end
                    if ColorInputConnection then ColorInputConnection:Disconnect() end
                    ColorInputConnection = RunService.RenderStepped:Connect(function()
                        local Pos = GetMousePosition()
                        local X = math.clamp((Pos.X - ColorSquare.AbsolutePosition.X) / ColorSquare.AbsoluteSize.X, 0, 1)
                        local Y = math.clamp((Pos.Y - ColorSquare.AbsolutePosition.Y) / ColorSquare.AbsoluteSize.Y, 0, 1)
                        ColorSelection.Position = UDim2.new(X, 0, Y, 0)
                        ColorS, ColorV = X, 1 - Y
                        UpdatePicker()
                    end)
                end
            end
            ColorSquare.InputBegan:Connect(OnColorSquareInput)
            ColorSquare.InputEnded:Connect(function(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    if ColorInputConnection then ColorInputConnection:Disconnect() end
                end
            end)

            local function OnHueStripInput(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    if RainbowMode then return end
                    if HueInputConnection then HueInputConnection:Disconnect() end
                    HueInputConnection = RunService.RenderStepped:Connect(function()
                        local Pos = GetMousePosition()
                        local Y = math.clamp((Pos.Y - HueStrip.AbsolutePosition.Y) / HueStrip.AbsoluteSize.Y, 0, 1)
                        HueSelection.Position = UDim2.new(0.5, 0, Y, 0)
                        ColorH = 1 - Y
                        UpdatePicker()
                    end)
                end
            end
            HueStrip.InputBegan:Connect(OnHueStripInput)
            HueStrip.InputEnded:Connect(function(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    if HueInputConnection then HueInputConnection:Disconnect() end
                end
            end)

            RainbowButton.MouseButton1Click:Connect(function()
                RainbowMode = not RainbowMode
                if ColorInputConnection then ColorInputConnection:Disconnect() end
                if HueInputConnection then HueInputConnection:Disconnect() end

                if RainbowMode then
                    TweenService:Create(RainbowButton, TweenInfo.new(0.2), {
                        BackgroundColor3 = Theme.Accent,
                        TextColor3 = Color3.new(1, 1, 1)
                    }):Play()
                    local OldColor = ColorDisplay.BackgroundColor3
                    local OldSquare = ColorSquare.BackgroundColor3
                    local OldSelPos = ColorSelection.Position
                    local OldHuePos = HueSelection.Position
                    local OldColorH, OldColorS, OldColorV = ColorH, ColorS, ColorV

                    while RainbowMode do
                        RainbowValue = (RainbowValue + 1/255) % 1
                        HuePosition = (HuePosition + 1) % 80
                        local NewColor = Color3.fromHSV(RainbowValue, 1, 1)
                        ColorDisplay.BackgroundColor3 = NewColor
                        ColorSquare.BackgroundColor3 = NewColor
                        ColorSelection.Position = UDim2.new(1, 0, 0, 0)
                        HueSelection.Position = UDim2.new(0.5, 0, 0, HuePosition)
                        if Props.Callback then Props.Callback(NewColor) end
                        Library:UpdateTheme(NewColor)
                        RunService.RenderStepped:Wait()
                    end

                    ColorDisplay.BackgroundColor3 = OldColor
                    ColorSquare.BackgroundColor3 = OldSquare
                    ColorSelection.Position = OldSelPos
                    HueSelection.Position = OldHuePos
                    ColorH, ColorS, ColorV = OldColorH, OldColorS, OldColorV
                    UpdatePicker()
                else
                    TweenService:Create(RainbowButton, TweenInfo.new(0.2), {
                        BackgroundColor3 = Theme.DarkContrast,
                        TextColor3 = Color3.fromRGB(200, 200, 200)
                    }):Play()
                end
            end)

            Trigger.MouseButton1Click:Connect(function()
                local Toggled = Row.Size.Y.Offset == 36
                if Toggled then
                    TweenService:Create(Row, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                        Size = UDim2.new(1, 0, 0, 160)
                    }):Play()
                else
                    TweenService:Create(Row, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                        Size = UDim2.new(1, 0, 0, 36)
                    }):Play()
                end
            end)
        end

        function Section:AddKeybind(Name, DefaultKey, OnActivate)
            local Row = CreateRow(Page, 36)
            CreateLabelInside(Row, Name, 10, 80)

            local BindButton = CreateInstance("TextButton", {
                Parent = Row,
                BackgroundColor3 = Theme.DarkContrast,
                BorderSizePixel = 0,
                Position = UDim2.new(1, -80, 0.5, -12),
                Size = UDim2.new(0, 70, 0, 24),
                Font = Enum.Font.GothamBold,
                Text = (DefaultKey and DefaultKey.Name) or "None",
                TextColor3 = Color3.fromRGB(200, 200, 200),
                TextSize = 12,
                AutoButtonColor = false,
            })
            AddCorner(BindButton, 4)
            CreateRipple(BindButton)

            local CurrentKey = DefaultKey
            local Listening = false
            BindButton.MouseButton1Click:Connect(function()
                if Listening then return end
                Listening = true
                BindButton.Text = "..."
                TweenService:Create(BindButton, TweenInfo.new(0.2), { TextColor3 = Theme.Accent }):Play()

                local Connection
                Connection = UserInputService.InputBegan:Connect(function(Input)
                    if Input.UserInputType == Enum.UserInputType.Keyboard then
                        CurrentKey = Input.KeyCode
                        BindButton.Text = CurrentKey.Name
                        TweenService:Create(BindButton, TweenInfo.new(0.2), { TextColor3 = Color3.fromRGB(200, 200, 200) }):Play()
                        Listening = false
                        Connection:Disconnect()
                    end
                end)
            end)

            UserInputService.InputBegan:Connect(function(Input, Processed)
                if not Processed and Input.KeyCode == CurrentKey then
                    if OnActivate then OnActivate() end
                end
            end)
        end

        return Section
    end
