local TweenInfo = TweenInfo.new

local function TweenObject(obj, properties, duration, easing, direction)
	TweenService:Create(obj, TweenInfo(duration or 0.2, easing or Enum.EasingStyle.Linear, direction or Enum.EasingDirection.In), properties):Play()
end

local function CreateCorner(parent, radius)
	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, radius or 4)
	corner.Parent = parent
	return corner
end

local function CloneWithRipple(button, sampleTemplate)
	local sample = sampleTemplate:Clone()
	sample.Parent = button
	local mouse = game:GetService("Players").LocalPlayer:GetMouse()
	local x, y = mouse.X - sample.AbsolutePosition.X, mouse.Y - sample.AbsolutePosition.Y
	sample.Position = UDim2.new(0, x, 0, y)
	local size = math.max(button.AbsoluteSize.X, button.AbsoluteSize.Y) * 1.5
	sample:TweenSizeAndPosition(
		UDim2.new(0, size, 0, size),
		UDim2.new(0.5, -size/2, 0.5, -size/2),
		Enum.EasingDirection.Out,
		Enum.EasingStyle.Quad,
		0.35,
		true
	)
	for i = 1, 10 do
		sample.ImageTransparency = sample.ImageTransparency + 0.05
		task.wait(0.35 / 12)
	end
	sample:Destroy()
end

local RegisteredElements = {}

local function RegisterElement(updateFn)
	table.insert(RegisteredElements, updateFn)
end

local function UpdateAllElements()
	for _, fn in ipairs(RegisteredElements) do
		fn()
	end
end

function Kavo:DraggingEnabled(frame, parent)
	parent = parent or frame
	local dragging = false
	local dragInput, mousePos, framePos

	frame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
			mousePos = input.Position
			framePos = parent.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)

	frame.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement then
			dragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			local delta = input.Position - mousePos
			parent.Position = UDim2.new(
				framePos.X.Scale, framePos.X.Offset + delta.X,
				framePos.Y.Scale, framePos.Y.Offset + delta.Y
			)
		end
	end)
end

local CurrentTheme = {}

function Kavo:ChangeColor(property, color)
	if property == "Background" then
		CurrentTheme.Background = color
	elseif property == "SchemeColor" then
		CurrentTheme.SchemeColor = color
	elseif property == "Header" then
		CurrentTheme.Header = color
	elseif property == "TextColor" then
		CurrentTheme.TextColor = color
	elseif property == "ElementColor" then
		CurrentTheme.ElementColor = color
	end
	UpdateAllElements()
end

				label.Font = Enum.Font.Gotham
				label.Text = "  " .. text
				label.RichText = true
				label.TextColor3 = CurrentTheme.TextColor
				label.TextSize = 14
				label.TextXAlignment = Enum.TextXAlignment.Left
				CreateCorner(label)
				if CurrentTheme.SchemeColor == Color3.fromRGB(255,255,255) then
					label.TextColor3 = Color3.fromRGB(0,0,0)
				elseif CurrentTheme.SchemeColor == Color3.fromRGB(0,0,0) then
					label.TextColor3 = Color3.fromRGB(255,255,255)
				end

				local function UpdateLabelColors()
					label.BackgroundColor3 = CurrentTheme.SchemeColor
					label.TextColor3 = CurrentTheme.TextColor
					if CurrentTheme.SchemeColor == Color3.fromRGB(255,255,255) then
						label.TextColor3 = Color3.fromRGB(0,0,0)
					elseif CurrentTheme.SchemeColor == Color3.fromRGB(0,0,0) then
						label.TextColor3 = Color3.fromRGB(255,255,255)
					end
				end
				RegisterElement(UpdateLabelColors)

				UpdateSectionSize()

				local LabelFunctions = {}
				function LabelFunctions:UpdateLabel(newText)
					label.Text = "  " .. newText
				end
				return LabelFunctions
			end

			return Elements
		end

		return Sections
	end

	return Tabs
end
