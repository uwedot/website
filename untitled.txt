local function MakeDraggable(Frame)
    local Dragging = false
    local DragStart, StartPos, DragInput

    local function UpdateDrag(Input)
        local Delta = Input.Position - DragStart
        Frame.Position = UDim2.new(
            StartPos.X.Scale, StartPos.X.Offset + Delta.X,
            StartPos.Y.Scale, StartPos.Y.Offset + Delta.Y
        )
    end

    Frame.InputBegan:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
            Dragging = true
            DragStart = Input.Position
            StartPos = Frame.Position
            Input.Changed:Connect(function()
                if Input.UserInputState == Enum.UserInputState.End then
                    Dragging = false
                end
            end)
        end
    end)

    Frame.InputChanged:Connect(function(Input)
        if Input.UserInputType == Enum.UserInputType.MouseMovement or Input.UserInputType == Enum.UserInputType.Touch then
            DragInput = Input
        end
    end)

    UserInputService.InputChanged:Connect(function(Input)
        if Input == DragInput and Dragging then
            UpdateDrag(Input)
        end
    end)
end

local function SetupHover(Button, NormalColor, HoverColor)
    Button.MouseEnter:Connect(function()
        TweenService:Create(Button, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageColor3 = HoverColor or Color3.fromRGB(255,255,255)}):Play()
    end)
    Button.MouseLeave:Connect(function()
        TweenService:Create(Button, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageColor3 = NormalColor or Color3.fromRGB(166,166,166)}):Play()
    end)
end

local function CreateCorneredFrame(Parent, Size, Position, Color, CornerRadius)
    local Frame = Instance.new("Frame")
    Frame.Parent = Parent
    Frame.BackgroundColor3 = Color or Color3.fromRGB(30,30,30)
    Frame.BorderSizePixel = 0
    if Size then Frame.Size = Size end
    if Position then Frame.Position = Position end
    local Corner = Instance.new("UICorner")
    Corner.CornerRadius = UDim.new(0, CornerRadius or 6)
    Corner.Parent = Frame
    return Frame
end

local function CreateListLayout(Parent, Padding, HorizontalAlignment)
    local Layout = Instance.new("UIListLayout")
    Layout.Parent = Parent
    Layout.HorizontalAlignment = HorizontalAlignment or Enum.HorizontalAlignment.Center
    Layout.SortOrder = Enum.SortOrder.LayoutOrder
    Layout.Padding = UDim.new(0, Padding or 5)
    return Layout
end

local function CreatePadding(Parent, Top, Left, Right, Bottom)
    local Pad = Instance.new("UIPadding")
    Pad.Parent = Parent
    if Top then Pad.PaddingTop = UDim.new(0, Top) end
    if Left then Pad.PaddingLeft = UDim.new(0, Left) end
    if Right then Pad.PaddingRight = UDim.new(0, Right) end
    if Bottom then Pad.PaddingBottom = UDim.new(0, Bottom) end
    return Pad
end

            Keybind.Text = ""
            Keybind.TextColor3 = Color3.fromRGB(0,0,0)
            Keybind.TextSize = 14
            local Corner = Instance.new("UICorner")

            Corner.Parent = Keybind

            local Label = Instance.new("TextLabel")
            Label.Name = "Title"

            Label.Size = UDim2.new(1, -6, 1, 0)

            Label.TextColor3 = Color3.fromRGB(255,255,255)
            Label.TextSize = 14

            local CurrentKeyLabel = Instance.new("TextLabel")
            CurrentKeyLabel.Name = "CurrentKey"
            CurrentKeyLabel.Parent = Keybind
            CurrentKeyLabel.AnchorPoint = Vector2.new(1, 0.5)
            CurrentKeyLabel.BackgroundColor3 = Color3.fromRGB(43,43,43)
            CurrentKeyLabel.Position = UDim2.new(1, -6, 0.5, 0)
            CurrentKeyLabel.Size = UDim2.new(-0, 46, 0, 24)
            CurrentKeyLabel.Font = Enum.Font.Gotham
            CurrentKeyLabel.Text = DefaultKey.Name or ". . ."
            CurrentKeyLabel.TextColor3 = Color3.fromRGB(255,255,255)
            CurrentKeyLabel.TextSize = 14
            local KeyCorner = Instance.new("UICorner")
            KeyCorner.CornerRadius = UDim.new(0, 4)
            KeyCorner.Parent = CurrentKeyLabel

            local Binding = false
            local CurrentKey = DefaultKey

            Keybind.InputBegan:Connect(function(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    Binding = true
                    CurrentKeyLabel.Text = ". . ."
                    local InputConnection
                    InputConnection = UserInputService.InputBegan:Connect(function(KeyboardInput)
                        if KeyboardInput.UserInputType == Enum.UserInputType.Keyboard then
                            CurrentKey = KeyboardInput.KeyCode
                            CurrentKeyLabel.Text = CurrentKey.Name
                            Binding = false
                            InputConnection:Disconnect()
                        end
                    end)
                end
            end)

            UserInputService.InputBegan:Connect(function(Input, GameProcessed)
                if not GameProcessed and Input.KeyCode == CurrentKey then
                    Callback(CurrentKey)
                end
            end)
        end

        function Elements:Dropdown(Text, Options, Callback, MultiSelect)
            MultiSelect = MultiSelect or false
            Options = Options or {}
            Callback = Callback or function() end

            local Dropdown = Instance.new("Frame")
            Dropdown.Name = "Dropdown"
            Dropdown.Parent = PageFrame
            Dropdown.BackgroundTransparency = 1
            Dropdown.ClipsDescendants = true
            Dropdown.Size = UDim2.new(1, -6, 0, 34)

            local DropdownList = CreateListLayout(Dropdown, 5, Enum.HorizontalAlignment.Center)

            local ChooseFrame = CreateCorneredFrame(Dropdown, UDim2.new(1, 0, 0, 34), nil, Color3.fromRGB(35,35,35))
            local TitleLabel = Instance.new("TextLabel")
            TitleLabel.Name = "Title"
            TitleLabel.Parent = ChooseFrame
            TitleLabel.BackgroundTransparency = 1
            TitleLabel.Position = UDim2.new(0, 8, 0, 0)
            TitleLabel.Size = UDim2.new(1, -6, 1, 0)
            TitleLabel.Font = Enum.Font.Gotham
            TitleLabel.Text = Text or "Dropdown"
            TitleLabel.TextColor3 = Color3.fromRGB(255,255,255)
            TitleLabel.TextSize = 14
            TitleLabel.TextXAlignment = Enum.TextXAlignment.Left

            local ArrowButton = Instance.new("ImageButton")
            ArrowButton.Name = "Arrow"
            ArrowButton.Parent = ChooseFrame
            ArrowButton.AnchorPoint = Vector2.new(1, 0.5)
            ArrowButton.BackgroundTransparency = 1
            ArrowButton.LayoutOrder = 10
            ArrowButton.Position = UDim2.new(1, -2, 0.5, 0)
            ArrowButton.Size = UDim2.new(0, 28, 0, 28)
            ArrowButton.ZIndex = 2
            ArrowButton.Image = "rbxassetid://3926307971"
            ArrowButton.ImageColor3 = Color3.fromRGB(161, 12, 59)
            ArrowButton.ImageRectOffset = Vector2.new(324, 524)
            ArrowButton.ImageRectSize = Vector2.new(36, 36)
            ArrowButton.ScaleType = Enum.ScaleType.Crop

            local OptionHolder = Instance.new("Frame")
            OptionHolder.Name = "OptionHolder"
            OptionHolder.Parent = Dropdown
            OptionHolder.BackgroundColor3 = Color3.fromRGB(35,35,35)
            OptionHolder.BorderSizePixel = 0
            OptionHolder.Position = UDim2.new(0, 0, 0, 34)
            OptionHolder.Size = UDim2.new(1, 0, 0, 0)
            local OptionCorner = Instance.new("UICorner")
            OptionCorner.CornerRadius = UDim.new(0, 6)
            OptionCorner.Parent = OptionHolder

            local OptionList = CreateListLayout(OptionHolder, 5, Enum.HorizontalAlignment.Center)
            CreatePadding(OptionHolder, 8)

            local Dropped = false
            local Selected = {}

            local function UpdateTitle()
                if MultiSelect then
                    local SelectedList = {}
                    for Item, _ in pairs(Selected) do
                        table.insert(SelectedList, Item)
                    end
                    if #SelectedList > 0 then
                        if #SelectedList <= 3 then
                            TitleLabel.Text = Text .. ": " .. table.concat(SelectedList, ", ")
                        else
                            TitleLabel.Text = Text .. ": " .. #SelectedList .. " selected"
                        end
                    else
                        TitleLabel.Text = Text
                    end
                end
            end

            local function CreateOption(OptionText)
                local OptionButton = Instance.new("TextButton")
                OptionButton.Name = "Option"
                OptionButton.Parent = OptionHolder
                OptionButton.BackgroundColor3 = Color3.fromRGB(134, 10, 49)
                OptionButton.BorderSizePixel = 0
                OptionButton.Size = UDim2.new(1, -16, 0, 30)
                OptionButton.AutoButtonColor = false
                OptionButton.Font = Enum.Font.Gotham
                OptionButton.Text = OptionText
                OptionButton.TextColor3 = Color3.fromRGB(255,255,255)
                OptionButton.TextSize = 14
                local OptCorner = Instance.new("UICorner")
                OptCorner.CornerRadius = UDim.new(0, 6)
                OptCorner.Parent = OptionButton

                if MultiSelect then
                    local Checkmark = Instance.new("ImageLabel")
                    Checkmark.Name = "Checkmark"
                    Checkmark.Parent = OptionButton
                    Checkmark.BackgroundTransparency = 1
                    Checkmark.Image = "rbxassetid://6031068421"
                    Checkmark.ImageTransparency = 1
                    Checkmark.Size = UDim2.new(0, 16, 0, 16)
                    Checkmark.Position = UDim2.new(0, 8, 0.5, -8)
                end

                local function OnSelect()
                    if MultiSelect then
                        if Selected[OptionText] then
                            Selected[OptionText] = nil
                            OptionButton.Checkmark.ImageTransparency = 1
                        else
                            Selected[OptionText] = true
                            OptionButton.Checkmark.ImageTransparency = 0
                        end
                        UpdateTitle()
                        local SelectedList = {}
                        for Item, _ in pairs(Selected) do
                            table.insert(SelectedList, Item)
                        end
                        Callback(SelectedList)
                    else
                        Callback(OptionText)
                        TitleLabel.Text = Text .. ": " .. OptionText
                        Dropped = false
                        TweenService:Create(ArrowButton, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Rotation = 0}):Play()
                        Dropdown:TweenSize(UDim2.new(1, -6, 0, 34), Enum.EasingDirection.Out, Enum.EasingStyle.Quart, 0.15, true)
                    end
                end
                OptionButton.InputBegan:Connect(function(Input)
                    if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                        OnSelect()
                    end
                end)
            end

            for _, Opt in ipairs(Options) do
                CreateOption(Opt)
            end

            OptionList:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
                if Dropped then
                    local Height = OptionList.AbsoluteContentSize.Y + 15
                    OptionHolder.Size = UDim2.new(1, 0, 0, Height)
                    Dropdown.Size = UDim2.new(1, -6, 0, 34 + Height)
                end
            end)

            local function ToggleDropdown()
                Dropped = not Dropped
                if Dropped then
                    local Height = OptionList.AbsoluteContentSize.Y + 15
                    Dropdown:TweenSize(UDim2.new(1, -6, 0, 34 + Height), Enum.EasingDirection.Out, Enum.EasingStyle.Quad, 0.15, true)
                    TweenService:Create(ArrowButton, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Rotation = 180}):Play()
                else
                    TweenService:Create(ArrowButton, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Rotation = 0}):Play()
                    Dropdown:TweenSize(UDim2.new(1, -6, 0, 34), Enum.EasingDirection.Out, Enum.EasingStyle.Quad, 0.15, true)
                end
            end
            ChooseFrame.InputBegan:Connect(function(Input)
                if Input.UserInputType == Enum.UserInputType.MouseButton1 or Input.UserInputType == Enum.UserInputType.Touch then
                    ToggleDropdown()
                end
            end)

            local DropdownMethods = {}
            function DropdownMethods:Refresh(NewOptions)
                NewOptions = NewOptions or {}
                Selected = {}
                Dropped = false
                TweenService:Create(ArrowButton, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Rotation = 0}):Play()
                Dropdown:TweenSize(UDim2.new(1, -6, 0, 34), Enum.EasingDirection.Out, Enum.EasingStyle.Quart, 0.15, true)

                for _, Child in ipairs(OptionHolder:GetChildren()) do
                    if Child:IsA("TextButton") then
                        Child:Destroy()
                    end
                end

                TitleLabel.Text = Text
                for _, Opt in ipairs(NewOptions) do
                    CreateOption(Opt)
                end
            end
            return DropdownMethods
        end

        return Elements
    end

    return TabFunctions
end
